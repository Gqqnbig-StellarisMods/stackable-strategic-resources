country_event = {
	id = stackableStrategicResources.1

	hide_window = yes

	mean_time_to_happen = { days = 30 }

	trigger = {
		has_country_strategic_resource = { type = sr_engos amount > 1 }
	}

	immediate = {
		if = {
			limit = {
				has_country_strategic_resource = { type = sr_engos amount = 2 }
			}
